CREATE TABLE `admin` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `login` varchar(40) NOT NULL,
  `mdps` text NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=MyISAM AUTO_INCREMENT=4 DEFAULT CHARSET=utf8



 CREATE TABLE `cours` (
  `id_cour` int(11) NOT NULL AUTO_INCREMENT,
  `nom` text NOT NULL,
  `description` text NOT NULL,
  `id_filiere` varchar(4) NOT NULL,
  PRIMARY KEY (`id_cour`),
  KEY `id_filiere` (`id_filiere`)
) ENGINE=MyISAM AUTO_INCREMENT=48 DEFAULT CHARSET=latin1


CREATE TABLE `demande` (
  `code_massar` int(11) NOT NULL AUTO_INCREMENT,
  `nom` varchar(25) NOT NULL,
  `prenom` varchar(25) NOT NULL,
  `date_naiss` date NOT NULL,
  `filiere` varchar(4) NOT NULL,
  `num_tele` int(11) NOT NULL,
  `email` varchar(30) NOT NULL,
  `mdps` text NOT NULL,
  `type_user` enum('etudiant','professeur') NOT NULL,
  `etat` enum('-1','0','1') NOT NULL,
  `adresse` text NOT NULL,
  PRIMARY KEY (`code_massar`),
  KEY `filiere` (`filiere`)
) ENGINE=MyISAM AUTO_INCREMENT=14 DEFAULT CHARSET=latin1



CREATE TABLE `etudiant` (
  `code_massar` int(11) NOT NULL AUTO_INCREMENT,
  `nom` varchar(25) DEFAULT NULL,
  `prenom` varchar(25) DEFAULT NULL,
  `date_naiss` date DEFAULT NULL,
  `filiere` varchar(4) DEFAULT NULL,
  `num_tele` int(11) DEFAULT NULL,
  `adresse` text,
  `email` varchar(30) DEFAULT NULL,
  `mdps` text NOT NULL,
  `type` varchar(10) DEFAULT 'etudiant',
  PRIMARY KEY (`code_massar`),
  KEY `filiere` (`filiere`)
) ENGINE=MyISAM AUTO_INCREMENT=29 DEFAULT CHARSET=latin1

CREATE TABLE `file` (
  `id_pdf` int(11) NOT NULL AUTO_INCREMENT,
  `pdf_lien` text NOT NULL,
  `id_filiere` varchar(4) NOT NULL,
  `code_prof` int(11) NOT NULL,
  `commantaire` text NOT NULL,
  `id_cour` int(11) DEFAULT NULL,
  `type_cour` enum('cour','tp') DEFAULT 'cour',
  PRIMARY KEY (`id_pdf`),
  KEY `id_filiere` (`id_filiere`),
  KEY `code_prof` (`code_prof`),
  KEY `id_cour` (`id_cour`)
) ENGINE=MyISAM AUTO_INCREMENT=3 DEFAULT CHARSET=latin1


CREATE TABLE `filiere` (
  `filiere_id` varchar(4) NOT NULL,
  `filiere` varchar(50) DEFAULT NULL,
  `filiere_description` text NOT NULL,
  PRIMARY KEY (`filiere_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1


CREATE TABLE `professeur` (
  `code_massar_prof` int(11) NOT NULL AUTO_INCREMENT,
  `nom` varchar(25) DEFAULT NULL,
  `prenom` varchar(25) DEFAULT NULL,
  `date_naiss` date DEFAULT NULL,
  `filiere` varchar(4) DEFAULT NULL,
  `num_tele` int(11) DEFAULT NULL,
  `email` varchar(30) DEFAULT NULL,
  `mdps` text NOT NULL,
  `type` varchar(10) DEFAULT 'professeur',
  `adresse` text,
  PRIMARY KEY (`code_massar_prof`),
  KEY `filiere` (`filiere`)
) ENGINE=MyISAM AUTO_INCREMENT=25 DEFAULT CHARSET=latin1


CREATE VIEW `login_etd` AS select `etudiant`.`code_massar` AS `code_massar`,concat(`etudiant`.`nom`,' ',`etudiant`.`prenom`) AS `nom_complet`,`etudiant`.`mdps` AS `mdps`,`etudiant`.`type` AS `type`,`etudiant`.`email` AS `email` from `etudiant`


CREATE VIEW `login_prof` AS select `professeur`.`code_massar_prof` AS `code_massar_prof`,concat(`professeur`.`nom`,' ',`professeur`.`prenom`) AS `nom_cmplt`,`professeur`.`email` AS `login`,`professeur`.`mdps` AS `mdps`,`professeur`.`type` AS `type` from `professeur`
